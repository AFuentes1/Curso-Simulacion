% ---------- DIAPOSITIVA: INTRODUCCIÓN ----------
\begin{frame}[shrink=10]{Introducción}
\begin{itemize}
  \item \textbf{Objetivo:} evaluar si las secuencias generadas se comportan como una \textbf{distribución uniforme} esperada.
  \item \textbf{Contexto:} un generador pseudoaleatorio debe producir valores \textit{sin patrones evidentes} y con frecuencias consistentes con la teoría.
  \item \textbf{Método:} aplicamos \textbf{pruebas estadísticas} para comparar resultados \textbf{observados vs. teóricos} en cada muestra.
  \item \textbf{Criterio:} nivel de significancia $\alpha=0.05$; si el estadístico cae en la \textbf{región de aceptación}, \textbf{no se rechaza} $H_0$.
  \item \textbf{Resultados:} se muestran \textbf{tablas y gráficos} y un resumen \textbf{PASA/FALLA} por lenguaje y prueba.
\end{itemize}
\end{frame}

\begin{frame}[shrink=5]{Conclusiones}
\begin{itemize}\itemsep6pt
  \item Para las muestras continuas $U(0,1)$ \textbf{(Java, Julia, Python y Erlang)}, los resultados fueron consistentes con uniformidad: \textbf{promedio, varianza, corridas, huecos (dígitos y números), póker y series} no rechazaron $H_0$ con $\alpha=0.05$.
  
  \item En las muestras discretas \textbf{Python $\{1..6\}$, C $\{1..4\}$, C $\{1..8\}$ y Racket $\{1..20\}$}, las pruebas aplicadas \textbf{(promedio, varianza y corridas)} no rechazaron $H_0$ con $\alpha=0.05$.

  \item Las pruebas de \textbf{huecos, póker y series} se enfocaron en las secuencias continuas $U(0,1)$ (tal como se trabajó en el análisis), mientras que para discretas se evaluaron principalmente \textbf{media/varianza} y \textbf{dependencia (corridas)}.

  \item En general, \textbf{no rechazar $H_0$} indica que \textbf{no hay evidencia estadística} (al 5\%) para decir que la secuencia no es uniforme/independiente; \textbf{no prueba} que el generador sea “perfecto”. Por eso es correcto usar \textbf{varias pruebas} para cubrir distribución e independencia.
\end{itemize}
\end{frame}
